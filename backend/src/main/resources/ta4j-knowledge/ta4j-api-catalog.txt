== COMMON STRATEGY PATTERNS ==

// KDJ Golden Cross / Death Cross Strategy:
StochasticOscillatorKIndicator k = new StochasticOscillatorKIndicator(series, 9);
StochasticOscillatorDIndicator d = new StochasticOscillatorDIndicator(k, 3);
Rule entryRule = new CrossedUpIndicatorRule(k, d);    // K crosses above D (golden cross)
Rule exitRule = new CrossedDownIndicatorRule(k, d);   // K crosses below D (death cross)

// MACD Strategy:
EMAIndicator shortEma = new EMAIndicator(closePrice, 12);
EMAIndicator longEma = new EMAIndicator(closePrice, 26);
MACDIndicator macd = new MACDIndicator(closePrice, 12, 26);
EMAIndicator signal = new EMAIndicator(macd, 9);
Rule entryRule = new CrossedUpIndicatorRule(macd, signal);
Rule exitRule = new CrossedDownIndicatorRule(macd, signal);

// RSI Strategy:
RSIIndicator rsi = new RSIIndicator(closePrice, 14);
Rule entryRule = new UnderIndicatorRule(rsi, 30);     // RSI below 30 (oversold)
Rule exitRule = new OverIndicatorRule(rsi, 70);       // RSI above 70 (overbought)

// Bollinger Bands Strategy:
SMAIndicator middle = new SMAIndicator(closePrice, 20);
BollingerBandsMiddleIndicator bbMiddle = new BollingerBandsMiddleIndicator(middle);
// Use BollingerBandFacade for easier construction:
BollingerBandFacade bb = new BollingerBandFacade(series, 20, 2);
Rule entryRule = new UnderIndicatorRule(closePrice, bb.lower());  // Price below lower band
Rule exitRule = new OverIndicatorRule(closePrice, bb.upper());    // Price above upper band

// Moving Average Crossover Strategy:
SMAIndicator shortSma = new SMAIndicator(closePrice, 5);
SMAIndicator longSma = new SMAIndicator(closePrice, 20);
Rule entryRule = new CrossedUpIndicatorRule(shortSma, longSma);
Rule exitRule = new CrossedDownIndicatorRule(shortSma, longSma);

// ADX Trend Strength Strategy:
ADXIndicator adx = new ADXIndicator(series, 14);
PlusDIIndicator plusDI = new PlusDIIndicator(series, 14);
MinusDIIndicator minusDI = new MinusDIIndicator(series, 14);
Rule entryRule = new AndRule(new OverIndicatorRule(adx, 25), new CrossedUpIndicatorRule(plusDI, minusDI));
Rule exitRule = new CrossedDownIndicatorRule(plusDI, minusDI);
